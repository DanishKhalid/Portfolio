Question:

Write a function that displays the marks range against a
particular grade. For example, if grade is “B”, then print Marks [
>= 60 and <70 ]

Code:

function marksRange ()
{
var grade =prompt("Enter your grade ") ;
switch (grade)
{
case "A1" :
alert ("Marks >=100 and <80");
break;
case "A" :
alert ("Marks >=79 and <80");
break;
case "B" :
alert ("Marks >= 60 and <70");
break;
case "C" :
alert ("Marks >=50 and <60");
break;
default :
alert ("You are Failed");
}
}
marksRange();
